Controls:
---------
    Mount:         right-click
    Dismount:      "E" + right-click or "E" + "S"
    Forward:       "W"
    Reverse:       "S"
    Left:          "A"
    Right:         "D"
    Up:            "Space"
    Down:          "Shift"
    Switch driver: "E" + "A"
